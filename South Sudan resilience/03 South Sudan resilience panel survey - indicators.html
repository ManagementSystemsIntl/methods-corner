Individual shocks

Code 
shk <- read_csv(here("output/tables/Shock incidence.csv"),
                show_col_types=F)

shk_gt <- shk %>%
  dplyr::select(Shock, Percent) %>%
  mutate(Bar=Percent*100) %>%
  gt() %>%
  gt_plt_bar_pct(column=Bar, fill=usaid_blue, background=light_grey, scaled=T) %>%
  cols_width(3 ~ px(125)) %>%
  fmt_percent(2, decimals=0) %>%
  cols_label(Bar="")
